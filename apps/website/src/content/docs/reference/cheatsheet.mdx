# Sruja Cheatsheet

## Elements

```sruja
specification {
  element person
  element system
  element container
  element component
  element datastore
  element queue
}

model {
  person User
  system App {
    container Web
    container API
    datastore DB
  }
  User -> App.Web "Uses"
  App.Web -> App.API "Calls"
  App.API -> App.DB "Reads/Writes"
}

views {
  view index {
    include *
  }
}
```

## Component

```sruja
specification {
  element system
  element container
  element component
}

model {
  App = system "App" {
    Web = container "Web" {
      Cart = component "Cart"
    }
  }
}
```

## Scenario

```sruja
specification {
  element person
  element system
  element container
  element datastore
}

model {
  User = person "User"
  
  App = system "App" {
    Web = container "Web"
    API = container "API"
    DB = datastore "Database"
  }
  
  scenario Checkout "Checkout Flow" {
    User -> App.Web "adds items"
    App.Web -> App.API "validates"
    App.API -> App.DB "stores order"
  }
}
```

## Deployment

```sruja
deployment Prod {
  node Cloud {
    node Region {
      node Service {
        containerInstance Web
      }
    }
  }
}
```
