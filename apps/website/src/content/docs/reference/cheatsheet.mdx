# Sruja Cheatsheet

## Elements

```sruja
person = kind "Person"
system = kind "System"
container = kind "Container"
component = kind "Component"
datastore = kind "Datastore"
queue = kind "Queue"

User = person "User"
App = system "App" {
Web = container "Web"
API = container "API"
DB = datastore "DB"
}
User -> App.Web "Uses"
App.Web -> App.API "Calls"
App.API -> App.DB "Reads/Writes"

view index {
include *
}
```

## Component

```sruja
system = kind "System"
container = kind "Container"
component = kind "Component"

App = system "App" {
Web = container "Web" {
  Cart = component "Cart"
}
}
```

## Scenario

```sruja
person = kind "Person"
system = kind "System"
container = kind "Container"
datastore = kind "Datastore"

User = person "User"

App = system "App" {
Web = container "Web"
API = container "API"
DB = datastore "Database"
}

scenario Checkout "Checkout Flow" {
User -> App.Web "adds items"
App.Web -> App.API "validates"
App.API -> App.DB "stores order"
}
```

## Deployment

```sruja
deployment Prod {
  node Cloud {
    node Region {
      node Service {
        containerInstance Web
      }
    }
  }
}
```
